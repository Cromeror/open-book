```typescript
// components/navigation/ResidentNav.tsx
'use client';

import Link from 'next/link';
import { usePathname } from 'next/navigation';
import {
  HomeIcon,
  CurrencyDollarIcon,
  DocumentTextIcon,
  ChatBubbleLeftRightIcon,
  UserCircleIcon,
} from '@heroicons/react/24/outline';

const navItems = [
  { href: '/resident/inicio', label: 'Inicio', icon: HomeIcon },
  { href: '/resident/objetivos', label: 'Objetivos', icon: CurrencyDollarIcon },
  { href: '/resident/mis-aportes', label: 'Mis Aportes', icon: DocumentTextIcon },
  { href: '/resident/pqr', label: 'PQR', icon: ChatBubbleLeftRightIcon },
  { href: '/resident/perfil', label: 'Mi Perfil', icon: UserCircleIcon },
];

interface ResidentNavProps {
  mobile?: boolean;
  onClose?: () => void;
}

export function ResidentNav({ mobile, onClose }: ResidentNavProps) {
  const pathname = usePathname();

  if (mobile) {
    return (
      <nav className="bg-white border-b shadow-lg">
        <ul className="py-2">
          {navItems.map((item) => {
            const isActive = pathname.startsWith(item.href);
            const Icon = item.icon;

            return (
              <li key={item.href}>
                <Link
                  href={item.href}
                  onClick={onClose}
                  className={`flex items-center px-4 py-3 ${
                    isActive ? 'bg-blue-50 text-blue-600' : 'text-gray-700'
                  }`}
                >
                  <Icon className="w-5 h-5 mr-3" />
                  {item.label}
                </Link>
              </li>
            );
          })}
        </ul>
      </nav>
    );
  }

  return (
    <nav className="hidden lg:flex items-center space-x-1">
      {navItems.map((item) => {
        const isActive = pathname.startsWith(item.href);

        return (
          <Link
            key={item.href}
            href={item.href}
            className={`px-4 py-2 rounded-lg text-sm font-medium transition-colors ${
              isActive
                ? 'bg-blue-100 text-blue-700'
                : 'text-gray-600 hover:bg-gray-100'
            }`}
          >
            {item.label}
          </Link>
        );
      })}
    </nav>
  );
}
```
