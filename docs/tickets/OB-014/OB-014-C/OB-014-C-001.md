## Notas de Implementacion

```typescript
// components/layouts/ResidentLayout.tsx
'use client';

import { useState } from 'react';
import { ResidentNav } from '@/components/navigation/ResidentNav';
import { ResidentHeader } from '@/components/navigation/ResidentHeader';

interface ResidentLayoutProps {
  children: React.ReactNode;
}

export function ResidentLayout({ children }: ResidentLayoutProps) {
  const [mobileMenuOpen, setMobileMenuOpen] = useState(false);

  return (
    <div className="min-h-screen bg-gray-50">
      <ResidentHeader
        onMenuToggle={() => setMobileMenuOpen(!mobileMenuOpen)}
      />

      {/* Mobile navigation */}
      {mobileMenuOpen && (
        <div className="lg:hidden">
          <ResidentNav mobile onClose={() => setMobileMenuOpen(false)} />
        </div>
      )}

      <main className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-8">
        {children}
      </main>
    </div>
  );
}

// app/(resident)/layout.tsx
import { ResidentLayout } from '@/components/layouts/ResidentLayout';
import { AuthGuard } from '@/components/guards/AuthGuard';

export default function ResidentRootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <AuthGuard>
      <ResidentLayout>{children}</ResidentLayout>
    </AuthGuard>
  );
}
```
