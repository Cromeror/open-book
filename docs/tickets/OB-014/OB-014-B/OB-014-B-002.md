```typescript
// components/navigation/AdminNav.tsx
'use client';

import Link from 'next/link';
import { usePathname } from 'next/navigation';
import {
  HomeIcon,
  UsersIcon,
  BuildingOfficeIcon,
  ChartBarIcon,
  CurrencyDollarIcon,
  DocumentTextIcon,
  Cog6ToothIcon,
} from '@heroicons/react/24/outline';

const navItems = [
  { href: '/admin/dashboard', label: 'Dashboard', icon: HomeIcon },
  { href: '/admin/usuarios', label: 'Usuarios', icon: UsersIcon },
  { href: '/admin/copropiedades', label: 'Copropiedades', icon: BuildingOfficeIcon },
  { href: '/admin/objetivos', label: 'Objetivos de Recaudo', icon: CurrencyDollarIcon },
  { href: '/admin/reportes', label: 'Reportes', icon: ChartBarIcon },
  { href: '/admin/auditoria', label: 'Auditoria', icon: DocumentTextIcon },
  { href: '/admin/configuracion', label: 'Configuracion', icon: Cog6ToothIcon },
];

interface AdminNavProps {
  collapsed: boolean;
}

export function AdminNav({ collapsed }: AdminNavProps) {
  const pathname = usePathname();

  return (
    <nav className="flex flex-col h-full">
      {/* Logo */}
      <div className="h-16 flex items-center justify-center border-b border-gray-800">
        {collapsed ? (
          <span className="text-white text-xl font-bold">OB</span>
        ) : (
          <span className="text-white text-xl font-bold">OpenBook</span>
        )}
      </div>

      {/* Nav items */}
      <ul className="flex-1 py-4 space-y-1">
        {navItems.map((item) => {
          const isActive = pathname.startsWith(item.href);
          const Icon = item.icon;

          return (
            <li key={item.href}>
              <Link
                href={item.href}
                className={`flex items-center px-4 py-3 text-sm transition-colors ${
                  isActive
                    ? 'bg-blue-600 text-white'
                    : 'text-gray-300 hover:bg-gray-800 hover:text-white'
                }`}
                title={collapsed ? item.label : undefined}
              >
                <Icon className="w-5 h-5 flex-shrink-0" />
                {!collapsed && (
                  <span className="ml-3">{item.label}</span>
                )}
              </Link>
            </li>
          );
        })}
      </ul>
    </nav>
  );
}
```
